# Environment variables (set these before running crontab):
SGL_BENCHMARK_CI_DIR="/mnt/raid/michael/sglang-ci"
GROK_MODEL_PATH="/data/grok-1-W4A8KV8"
GROK2_MODEL_PATH="/data2/grok-2/"

# 7:30 AM PT (9:30 AM CT) - Docker image availability check with Teams notifications
30 9 * * * cd "$SGL_BENCHMARK_CI_DIR" && TEAMS_WEBHOOK_URL="$TEAMS_WEBHOOK_URL" bash nightly_image_check.sh > cron/cron_log/docker_image_check_$(date +\%Y\%m\%d).log 2>&1

0 10 * * * cd "$SGL_BENCHMARK_CI_DIR" && bash perf_nightly.sh --model=grok2 --model-path="$GROK2_MODEL_PATH" --mode=online --hardware="$HARDWARE_TYPE" --teams-webhook-url="$TEAMS_WEBHOOK_URL" > cron/cron_log/grok2_nightly_online_$(date +\%Y\%m\%d).log 2>&1

# 10 AM PT (12 PM CT) - Grok online and offline benchmark with gating logic + Teams notifications

# 12:30 PM PT (2:30 PM CT) - DeepSeek online serving benchmark + Teams notifications
30 14 * * * cd "$SGL_BENCHMARK_CI_DIR" && bash perf_nightly.sh --model=deepseek --model-path="$DEEPSEEK_MODEL_PATH" --mode=online --hardware="$HARDWARE_TYPE" --teams-webhook-url="$TEAMS_WEBHOOK_URL" > cron/cron_log/deepseek_nightly_online_$(date +\%Y\%m\%d).log 2>&1
